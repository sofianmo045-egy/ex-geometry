(previous functions: getV, format, dist3D, getP, openTab, saveHistory, showHistoryByActiveTab, toggleHistory, calcMeasurements) function calcCalibration() { const BP = getP('bp'), ST = getP('st'), BU = getP('bu'), S3 = getP('s3'), S4 = getP('s4'), S5 = getP('s5'), T1 = getP('t1'); const s1 = (ST && BU) ? dist3D(ST, BU) : null; let s3_val = null, s2_val = null, t6_val = null, bsv = null; if (BP && ST) bsv = ST.z - BP.z; if (s1 && S3) { const v_st_bu = { x: BU.x - ST.x, y: BU.y - ST.y, z: BU.z - ST.z }; const v_st_s3 = { x: S3.x - ST.x, y: S3.y - ST.y, z: S3.z - ST.z }; // S3: Perpendicular distance from S3 to the Stick (ST-BU line) const cp = { x: v_st_s3.y * v_st_bu.z - v_st_s3.z * v_st_bu.y, y: v_st_s3.z * v_st_bu.x - v_st_s3.x * v_st_bu.z, z: v_st_s3.x * v_st_bu.y - v_st_s3.y * v_st_bu.x }; s3_val = Math.sqrt(cp.x**2 + cp.y**2 + cp.z**2) / s1; // S2: Distance along the stick from Stick Pin (ST) to the projection of S3 const dotProduct = (v_st_s3.x * v_st_bu.x + v_st_s3.y * v_st_bu.y + v_st_s3.z * v_st_bu.z); s2_val = dotProduct / s1; } if (T1 && BU && S5) { const vT1 = { x: T1.x - BU.x, y: T1.y - BU.y, z: T1.z - BU.z }; const vS5 = { x: S5.x - BU.x, y: S5.y - BU.y, z: S5.z - BU.z }; const m1 = Math.sqrt(vT1.x**2 + vT1.y**2 + vT1.z**2), m2 = Math.sqrt(vS5.x**2 + vS5.y**2 + vS5.z**2); if (m1 > 0 && m2 > 0) { t6_val = Math.acos(Math.max(-1, Math.min(1, (vT1.x * vS5.x + vT1.y * vS5.y + vT1.z * vS5.z) / (m1 * m2)))) * (180 / Math.PI); } } const res = { 'out-boomL_cal': format(dist3D(BP, ST), " m"), 'out-stickL_cal': format(s1, " m"), 'out-bsv': format(bsv, " m"), 'out-s2': format(s2_val, " m"), 'out-s3': format(s3_val, " m"), 'out-s4': format(dist3D(S3, S4), " m"), 'out-s5': format(dist3D(S4, S5), " m"), 'out-ts': format(dist3D(BU, S5), " m"), 'out-t1': format(dist3D(T1, BU), " m"), 'out-t6': format(t6_val, "°") }; for (let id in res) { const el = document.getElementById(id); if(el) { el.innerHTML = res[id]; localStorage.setItem('res_'+id, res[id]); } } document.getElementById('resultsCalibration').style.display = "block"; localStorage.setItem('show_res2', 'true'); saveHistory('calibration', res); } // --- Export and Download Logic --- function exportReport2() { const data = [ { label: 'GEOMETRY LENGTHS', isHeader: true }, { label: 'Boom Length', value: document.getElementById('out-boomL_cal').innerText }, { label: 'Stick Length', value: document.getElementById('out-stickL_cal').innerText }, { label: 'Boom Stick V (ΔZ)', value: document.getElementById('out-bsv').innerText }, { label: 'SUPPORT LEVER ATTACHMENT', isHeader: true }, { label: 'S2', value: document.getElementById('out-s2').innerText }, { label: 'S3', value: document.getElementById('out-s3').innerText }, { label: 'S4', value: document.getElementById('out-s4').innerText }, { label: 'S5', value: document.getElementById('out-s5').innerText }, { label: 'BUCKET & TEETH', isHeader: true }, { label: 'T5 (Holes separation)', value: document.getElementById('out-ts').innerText }, { label: 'T1', value: document.getElementById('out-t1').innerText }, { label: 'T6 (Tooth angle)', value: document.getElementById('out-t6').innerText } ]; const html = generateHTMLReport('Crane Configuration Report', data); downloadFile(html, 'Crane_Configuration_Report.doc'); } // Initialize Auto-save/load document.querySelectorAll('.saveable').forEach(input => { input.addEventListener('input', autoSave); }); window.onload = autoLoad;
